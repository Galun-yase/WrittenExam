    for ("遍历本层集合中元素") {
        // 一些逻辑操作（可有可无，视情况而定）

        // 5、做出选择
        // 6、递归调用该函数，进入下一层继续搜索
        backtrack("新的参数");

        // 7、撤销选择，回到上一层的状态
